Audit Internal Only

Request audit logging and compliance records for enterprise customers.

⚠️ Internal Service

The Audit system is an internal service with no public REST API. Audit logs are automatically captured by middleware and stored for compliance purposes.

Audit Middleware

The AuditMiddleware automatically logs all API requests with relevant metadata. 

# apps/audit/middleware.py

class AuditMiddleware:
    """Logs all API requests for compliance and debugging."""
    
    def __call__(self, request):
        # Capture request metadata
        audit_entry = {
            "timestamp": timezone.now(),
            "user_id": request.user.id if request.user.is_authenticated else None,
            "org_id": get_current_org_id(request),
            "method": request.method,
            "path": request.path,
            "ip_address": get_client_ip(request),
            "user_agent": request.META.get("HTTP_USER_AGENT"),
            "request_id": request.META.get("HTTP_X_REQUEST_ID"),
        }
        
        response = self.get_response(request)
        
        # Add response metadata
        audit_entry["status_code"] = response.status_code
        audit_entry["response_time_ms"] = calculate_response_time()
        
        # Async write to avoid blocking
        create_audit_log.delay(audit_entry)
        
        return response
