Rule Data Model

{
  "id": "rule-uuid",
  "name": "Auto-approve posts from admins",
  "description": "Automatically approve posts created by admin users",
  "trigger": {
    "type": "post.created",
    "conditions": [
      { "field": "author.role", "operator": "in", "value": ["admin", "owner"] }
    ]
  },
  "actions": [
    { "type": "post.approve" },
    { "type": "notification.send", "params": { "message": "Post auto-approved" } }
  ],
  "is_enabled": true,
  "execution_count": 42,
  "last_executed_at": "2024-01-15T10:30:00Z",
  "created_at": "2024-01-01T00:00:00Z"
}
